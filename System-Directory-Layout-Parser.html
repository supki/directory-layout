Safe HaskellNone

System.Directory.Layout.Parser

Description

Parser for text format for DL data structure, for example 

 c/
 ..x
 ..y
 ....n
 ....n
 ....
 ..z
 ....t
 ....t
 ....
 d/

where . stands for space is equivalent of 

 do directory "c" $ do
      file_ "x"
      file "y" "nnnn"
      file "z" "tntn"
    directory_ "d"
